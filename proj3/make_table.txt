use trst019;

create table products(
			sku char(9) PRIMARY KEY NOT NULL,
			title varchar(25) NOT NULL,
			quantity int(3) NOT NULL,
			cost decimal(6,2) NOT NULL,
			first_name varchar(25) NOT NULL,
    		last_name varchar(25) NOT NULL,
			address varchar(50) NOT NULL,
			city varchar(20) NOT NULL,
			state varchar(2) NOT NULL,
			zipcode int(5) NOT NULL,
    		areaphone varchar(3) NOT NULL,
			prefixphone varchar(3) NOT NULL,
			phone varchar (4) NOT NULL,
			first_nametwo varchar(25) NOT NULL,
    		last_nametwo varchar(25) NOT NULL,
			addresstwo varchar(50) NOT NULL,
			citytwo varchar(20) NOT NULL,
			statetwo varchar(2) NOT NULL,
			zipcodetwo int(5) NOT NULL,
    		areaphonetwo varchar(3) NOT NULL,
			prefixphonetwo varchar(3) NOT NULL,
			phonetwo varchar (4) NOT NULL,
			payment varchar(20) NOT NULL,														
			car_number varchar(20) NOT NULL,
			date numeric(2) NOT NULL,							
			year numeric(4) NOT NULL);
